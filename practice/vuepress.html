# Markdown 문서 Merge

  • 표지용 READEME.md 와 Contents용 manual.md 로 구분한다.
  • manual.md는 흩어져 있는 manual page를 merge.sh(자작)를 실행시켜 순서대로 합친다.\
    • plugin은 기본적으로 page 순서를 보장하지 않음
#!/bin/sh 

# 병합할 문서들을 순서대로 기술한다.
list='toc.md
job-properties.md
job-schedule.md
'
# 출력파일 명 지정
> manual.md
for md in $list
do
    echo $md
    if [[ $md = toc.md ]]; then
        cat $md >> manual.md
    else
        # 개별 문서의 목차 제거
        sed '/\[toc\]/d' $md | sed '/# Table/d' >> manual.md
    fi
done
